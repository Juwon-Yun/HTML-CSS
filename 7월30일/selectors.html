script에서 html요소에 접근할 때
    id를 이용해서 접근
            - document.getElemnetById('id') //더 효율적이다. es2버전
            - document.querySelector('#id') //new 버전, es6버전

    tag를 이용해서 접근
            - document.getElementsByTagsName('tag명')[n]    //n번째 태그 선택
            - document.getElementsByTagsName('tag명')       //모든 'tag명' 요소 선택
            - document.querySelector('tag명')               //첫 번째 태그만 선택
            - document.querySelectorAll('tag명')            //모든 'tag명' 요소 선택
    
    class를 이용해서 접근
            - document.getElementsByClassName('class명')[n] //n번째 클래스 요소 선택    
            - document.getElementsByClassName('class명')    //모든 'class명' 요소 선택
            - document.querySelector('.class명')            //첫 번째 'class명' 요소만 선택
            - document.querySelectorAll('.class명')         //모든 'class명' 요소 선택
